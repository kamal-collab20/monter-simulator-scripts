HUD & Visual Enhancement Script

This additional script ensures a clean visual experience by removing blur effects and keeping the game HUD always visible. Ideal for players who want maximum clarity and performance.

-- ðŸ”¥ Simple Script: Removes Blur & Keeps HUD Enabled ðŸ”¥
local Players = game:GetService("Players")
local Lighting = game:GetService("Lighting")
local player = Players.LocalPlayer
local PlayerGui = player:WaitForChild("PlayerGui")

-- Function to remove any Blur effects
local function removeBlur()
    for _, effect in pairs(Lighting:GetChildren()) do
        if effect:IsA("BlurEffect") then
            effect:Destroy()
        end
    end
end

-- Function to enforce HUD visibility
local function fixHUD()
    for _, gui in pairs(PlayerGui:GetChildren()) do
        if gui:IsA("ScreenGui") then
            gui.Enabled = true
        end
    end
end

-- Continuous loop to maintain HUD and remove blur
while true do
    removeBlur()
    fixHUD()
    task.wait(0.5)
end
